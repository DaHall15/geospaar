How about SpatialPolygons?

Answers

library(sp)

# spatial points
coordinates(pts) <- ~x + y

# spatial lines
line1 <- as(pts, "SpatialLines")

# Another line
line2 <- as(pts[c(2, 1, 3, 4), ], "SpatialLines")

# Polygons
poly1 <- as(line1, "SpatialPolygons")  # fail
## Error in as(line1, "SpatialPolygons"): no method or default for coercing "SpatialLines" to "SpatialPolygons"
poly1 <- as(pts, "SpatialPolygons")  # fail
## Error in as(pts, "SpatialPolygons"): no method or default for coercing "SpatialPoints" to "SpatialPolygons"
# Correct
p <- Polygon(pts)
ps <- Polygons(list(p), 1)
sps <- SpatialPolygons(list(ps))

plot(sps, col = "grey")
lines(line1, col = "red")
lines(line2, col = "blue")
points(pts)
